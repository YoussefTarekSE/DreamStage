# ── Real-time pitch detection (production-safe, ~80 MB RAM) ──────────────────
# CREPE: Convolutional Representation for Pitch Estimation
# Kim et al. (2018) ISMIR. MIT license.
# Model: tiny (14M params), downloads on first use (~15 MB weights)
crepe>=0.0.15

# ── Training infrastructure ───────────────────────────────────────────────────
scikit-learn>=1.3.0        # RandomForest, GradientBoosting, cross-validation
pandas>=2.0.0              # feature DataFrames, Parquet I/O
pyarrow>=12.0.0            # Parquet backend (required by pandas)
joblib>=1.3.0              # model serialization (also used by sklearn)
matplotlib>=3.7.0          # learning curves, confusion matrices

# ── Local MusicGen (development / high-RAM machines only) ─────────────────────

# Facebook AudioCraft — MusicGen, AudioGen, EnCodec
# Copet et al. (2023) NeurIPS. MIT license.

# Note: audiocraft pins specific torch versions — let it install its own
